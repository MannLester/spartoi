"""
Spartoi Primitive: FastAPI API Key Authenticator
Universal, zero-dependency header injection security gate.
"""
import os
from fastapi import Security, HTTPException, status
from fastapi.security.api_key import APIKeyHeader

API_KEY_NAME = "X-Spartoi-API-Key"
api_key_header = APIKeyHeader(name=API_KEY_NAME, auto_error=False)

async def APIKeySecurity(api_key_header: str = Security(api_key_header)):
    # Pull expected master key token directly from native execution environment variables
    expected_key = os.getenv("SPARTOI_API_KEY")
    
    if not expected_key:
        raise HTTPException(
            status_code=status.HTTP_500_INTERNAL_SERVER_ERROR,
            detail="Server Security Misconfiguration: SPARTOI_API_KEY environment token is unset."
        )
        
    if api_key_header == expected_key:
        return api_key_header
        
    raise HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Invalid or missing API key signature."
    )
